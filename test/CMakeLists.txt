cmake_minimum_required(VERSION 3.8)
set(this InternetMessageTests)

set(Sources 
    src/InternetMessageTests.cpp
)

add_executable(${this} ${Sources})
set_target_properties(${this} PROPERTIES
    FOLDER Tests
)

target_include_directories(${this} PRIVATE ..)

    gtest_main
    InternetMessage
)

    NAME ${this}
    COMMAND ${this}
)
